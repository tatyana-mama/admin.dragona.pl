Интеграция с сайтом

Добавьте этот скрипт в файл dragona_main.html, чтобы заявки отправлялись в Telegram.

function submitForm() {
  const name = document.querySelector('input[placeholder="Anna Kowalska"]').value;
  const phone = document.querySelector('input[type="tel"]').value;
  const email = document.querySelector('input[type="email"]').value;
  const workshop = document.querySelector('select').value;
  const date = document.querySelector('input[type="date"]').value;
  const persons = document.querySelector('input[type="number"]').value;
  const message = document.querySelector('textarea').value;

  const TOKEN = 'ВАШ_BOT_TOKEN';
  const CHAT_ID = 'ВАШ_CHAT_ID';
  const text = `📩 Nowa zaявka!\n\n👤 ${name}\n📱 ${phone}\n✉️ ${email}\n🏺 ${workshop}\n📅 ${date}\n👥 ${persons} os.\n💬 ${message}`;

  fetch(`https://api.telegram.org/bot${TOKEN}/sendMessage`, {
    method: 'POST',
    headers: {'Content-Type':'application/json'},
    body: JSON.stringify({chat_id: CHAT_ID, text, parse_mode: 'HTML'})
  });
  alert('Dziękujemy! Potwierdzimy rezerwację w ciągu 24h.');
}
